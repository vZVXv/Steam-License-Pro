Как выгрузить лицензии из Steam

  1. Откройте Лицензии и активации ключей в браузере.
  2. Нажмите F12 (или ПКМ -> Исследовать элемент) и откройте вкладку Console.
  3. Вставьте код ниже и нажмите Enter. Он автоматически соберет все лицензии на странице.
  4. Перетащите скачанный файл steam_licenses.json в окно ниже.
(function() { // Находим все строки таблицы с атрибутом data-panel const rows = document.querySelectorAll('tr[data-panel]'); const licenses = []; rows.forEach(row => { // Ищем нужные колонки внутри строки const dateEl = row.querySelector('.license_date_col'); const methodEl = row.querySelector('.license_acquisition_col'); const tds = row.querySelectorAll('td'); // Проверяем, что нужные элементы существуют и колонок не меньше трех if (dateEl && methodEl && tds.length >= 3) { const date = dateEl.innerText.trim(); const name = tds[1].innerText.trim(); // Название всегда во второй колонке const method = methodEl.innerText.trim(); licenses.push({ date: date, name: name, method: method }); } }); // Если ничего не найдено, выводим предупреждение if (licenses.length === 0) { console.warn("Лицензии не найдены! Убедитесь, что вы находитесь на нужной странице."); return; } // Преобразуем массив объектов в красиво отформатированный JSON const jsonString = JSON.stringify(licenses, null, 2); // Создаем Blob (файл) из строки const blob = new Blob([jsonString], { type: "application/json" }); const url = URL.createObjectURL(blob); // Создаем невидимую ссылку для скачивания const a = document.createElement('a'); a.href = url; a.download = "steam_licenses.json"; // Имя итогового файла // Эмулируем клик для старта загрузки document.body.appendChild(a); a.click(); // Убираем за собой document.body.removeChild(a); URL.revokeObjectURL(url); console.log(`✅ Сбор завершен! Найдено лицензий: ${licenses.length}. Скачивание файла началось...`); })();
